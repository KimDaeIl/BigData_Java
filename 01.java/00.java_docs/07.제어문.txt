
## 제어문 (Flow control)

	-- 문장 수행의 흐름을 제어
	-- 조건문
	-- 반복문
	-- 분기문
	
## 조건문 : 조건에 따라 다르게 수행

1.	
		if(조건문) { 
			true 수행문; 
		}
		
2. 	
		if(조건문) { 
			true 수행문; 
		} else {
			false 수행문;
		}	

3.		
		if(조건문1) { 
			조건문1 true 수행문; 
		} else (조건문2) {
			조건문2 true 수행문;
		} else (조건문x) {
			조건문x true 수행문;
		}	
		
4.		
		if(조건문1) { 
			조건문1 true 수행문; 
		} else (조건문2) {
			조건문2 true 수행문;
		} else (조건문x) {
			조건문x true 수행문;
		}	else {
			해당조건이 없는 경우 수행문(기본수행문);
		}		
		
## 다중 조건문	
		switch(변수명 또는 수식) {
			case 값1:
				값1인 경우 수행문;
				[ break; ]
			case 값2:
				값2인 경우 수행문;
				break;
			case 값x:
				값x인 경우 수행문;
				break;
			default:
				기본 수행문;			
		}					
		
## 반복문 : 조건이 true인 동안 반복적으로 수행

1. 	먼저 조건비교후 수행결정: 
		while(조건문) {
			수행문;
		}
		
		
2.	먼저 한번 수행후 조건비교 반복 수행여부 결정: 최소 한번 수행
		do {
			수행문;
		} while(조건문);		
		
3. 	
		for(초기값; 조건문; 증감값) {
			수행문;
		}
		
		초기값: 0 ~ n, 조건문: 0,1, 증감값: 0 ~ n
		
## 분기문
1. break;
		-- switch, loop 반복문을 탈출
		
2. continue;
		-- loop 반복문의 조건문으로 분기하여, 조건 비교후 반복수행여부 결정
		
3. return;		
		-- 메서드 내부에서 선택, 필수 사용
		-- 메서드 호출측으로 분기(돌아감)
		-- 메서드의 반환타입이 있는 경우에는 필수 
		-- 방법: return; 또는 return 반환값;
